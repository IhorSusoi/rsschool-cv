```
function nthFibo(n) {
  let a = 1;
  let b = 1;
  let c = a + b;
  let j_it = 4;
  if(n == 1) return 0;
  if(n == 2 || n == 3) return 1;
  for (let i = 4; i < n; i++ ){
    a = b;
    b = c;
    c = a + b;
    if(i == c) j_it++;
  }
  return c;
}
```
